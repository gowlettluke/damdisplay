var structure = null; //holds structure.json data
var postDataBrokerLoadCallbacks = [];
var stationDataList = {}; //holds all of the stations and their data

async function databrokerStartup()
{
    stationDataList = {};
    structure       = await getSiteStructureJSON();

    //first load all the sites found in structure.json file
    await loadSitesFromStructureJsonFile();

    //only load all dams from data broker if one of the controls on the page requires it
    if (jQuery(".databroker-require-all-dams").length > 0)
    {
        console.log("Loading all dams");
        await loadAllDams();
    }

    if (postDataBrokerLoadCallbacks.length > 0)
    {
        for (let index = 0; index < postDataBrokerLoadCallbacks.length; index++) {
            const func = postDataBrokerLoadCallbacks[index];
            func();
        }
    }
    
}


// Load a single station code and populate the UI with the properties for that station
// Used on single dam display screens
function loadDataForSingleStationAfterReady(stationCode){

    jQuery(document).ready(function(){
        loadStation(stationCode);       

    });
}


async function loadAllDams()
{


    //now load the meta data for each site from data broker
    let stationCodes = [];

    for (var k in stationDataList) {        
        if (k){
            const s = stationDataList[k];
            stationCodes.push(k);    
        }
    }

    const sites = await getSites(stationCodes);


    for (let index = 0; index < sites.length; index++) {
        const d = sites[index];        
        stationDataList[d.station].stationCode    = d.station;
        stationDataList[d.station].stationDetails = d;
        stationDataList[d.station].stats = {};
        stationDataList[d.station].bga = {};
    }
    
    //finally load current storage capacities
    let currentStorageCapacities = await getCurrentStorageCapacityForStations(stationCodes);
    for (let index = 0; index < currentStorageCapacities.length; index++) {
        const currentCapacity = currentStorageCapacities[index];
        stationDataList[currentCapacity.station].currentCapacity = currentCapacity;        
    }


    populateUI();
    

}

function populateUI()
{
    //update display data for each site loaded
    // show the values stored
    for (var k in stationDataList) {
        // use hasOwnProperty to filter out keys from the Object.prototype
        if (stationDataList.hasOwnProperty(k)) 
        {
            const s = stationDataList[k];
            populatePropertiesForStation
            (
                {
                    stationCode     : s.stationCode,
                    currentCapacity : s.currentCapacity,
                    stats           : s.stats,
                    stationDetails  : s.stationDetails,
                    bga             : s.bga,
                    metaData        : s.metaData
                }        
            ); 
        }
    }

}

async function loadSitesFromStructureJsonFile()
{

    console.log("-- Start loading structure from json file");

    //enrich the station data list with data from the structure.json file
    for (let regionIndex = 0; regionIndex < structure.length; regionIndex++) {
        const regionNodeData = structure[regionIndex];
        const regionName = regionNodeData.region.name;
        console.log("Loading data for region " + regionName);

        for (let schemeIndex = 0; schemeIndex < regionNodeData.schemes.length; schemeIndex++) {
            const schemeData = regionNodeData.schemes[schemeIndex];

            for (let storageIndex = 0; storageIndex < schemeData.storages.length; storageIndex++) {
                
                const storageData = schemeData.storages[storageIndex];

                if (storageData.station){

                    console.log("loading station " + storageData.station + " from storages");
               
                    if (!stationDataList.hasOwnProperty(storageData.station)){
                        stationDataList[storageData.station] = {};
                    }
                    
                    stationDataList[storageData.station].metaData = 
                    {
                        location  : storageData.location,
                        siteType  : storageData.siteType,
                        station   : storageData.station,
                        shortName : storageData.shortName,
                        schemeName: schemeData.name,                                                
                        regionName: regionNodeData.region.name,
                        isDam     : false
                        
                    };
                }


            }

            for (let storageIndex = 0; storageIndex < schemeData.dams.length; storageIndex++) {
                const storageData = schemeData.dams[storageIndex];
               
                if (storageData.station)
                {

                    console.log("loading station " + storageData.station + " from dams");
               
                    if (!stationDataList.hasOwnProperty(storageData.station))
                    {
                        stationDataList[storageData.station] = {};
                    }

                    stationDataList[storageData.station].metaData = 
                    {
                        location  : storageData.location,
                        siteType  : storageData.siteType,
                        station   : storageData.station,
                        shortName : storageData.shortName,
                        schemeName: schemeData.name,
                        regionName: regionNodeData.region.name,
                        isDam     : true
                        
                    };
                }

            }


        }
       
    } 


    console.log("-- End loading structure from json file");
}




async function loadStation(stationCode)
{

    if (!stationCode){
        console.error("No station code provided");
        return;
    }                


    let currentCapacity = null;
    let stats           = null;
    let stationDetails  = null;
    let bga             = null;

    currentCapacity = await getCurrentStorageCapacity(stationCode);
    stats           = await getStationStatistics     (stationCode);
    stationDetails  = await getStationDetails        (stationCode);
    bga             = await getStationBGA            (stationCode);
    
    stationDataList[stationCode].stationCode     = stationCode;
    stationDataList[stationCode].stationDetails  = stationDetails;
    stationDataList[stationCode].stats           = stats;
    stationDataList[stationCode].bga             = bga;
    stationDataList[stationCode].currentCapacity = currentCapacity;
    
    await populateUI();


}

function populatePropertiesForStation(data){

    jQuery('.databroker-field').each(function() {

        var self            = this;
        var $self           = jQuery(self);
        var attrStationCode = $self.attr('data-databroker-stationcode');
        var attrFieldName   = $self.attr('data-databroker-field');

        if (attrStationCode == data.stationCode){

            switch (attrFieldName) {
                case 'percentageFull':
                    $self.text(data.currentCapacity.percentageFull + "%");
                    break;
                case 'totalCapacityMegaLitres':
                    $self.text(data.currentCapacity.totalCapacityMegaLitres);
                    break;
                case 'historicalLowStorageCapacity':
                    $self.text(data.stats.historicalLowStorageCapacity + "%");
                    break;
                case 'historicalHighStorageCapacity':
                    $self.text(data.stats.historicalHighStorageCapacity + "%");
                    break;
                case 'historicalHighStorageCapacityDate':                  
                    $self.text(friendlyDate(new Date(data.stats.historicalHighStorageCapacityDate)));
                    break;            
                case 'historicalLowStorageDate':                
                    $self.text(friendlyDate(new Date(data.stats.historicalLowStorageDate)));
                    break;
                case 'percentageFull-time':                
                    $self.text(friendlyDateFull(new Date(data.currentCapacity.date)));
                    break;            
                case 'siteShortName':
                    $self.text(data.stationDetails.shortName);    
                    break;
                case 'siteLocation':
                    if (data.hasOwnProperty('metaData')){
                        $self.text(data.metaData.location);    
                    }
                    break;
                case 'siteSchemeName':
                    if (data.hasOwnProperty('metaData'))
                    {
                        if (data.metaData.hasOwnProperty('schemeName')){
                            $self.text(data.metaData.schemeName);    
                        }
                        else 
                        {
                            $self.text("Site " + data.stationCode);    
                        }
                    }
                    break;
                case 'hazardDescription':
                    let hazardDescription = data.bga.hazardStatus;                
                    $self.text(hazardDescription);    
                    break;
                case 'bgaImage':
                    let imgPath = "/wp-content/themes/sunwater/images/";

                    switch(data.bga.actionLevel)
                    {
                        case 1:
                            imgPath += "BGA_Low.png";
                            break;
                        case 2:
                            imgPath += "BGA_Medium.png";
                            break;
                        case 3:
                            imgPath += "BGA_High.png";
                            break;
                        default:
                            imgPath += "BGA_NoData.png";
                            break;

                    }
                    $self.attr('src', imgPath);
                    break;
                case 'damNavigationUrl':

                    let navUrl = "/dams/" +  data.metaData.shortName.replace(' ', '-');
                    
                    $self.attr('href', navUrl);
                    break;
                default:
                // do nothing
                break;
            }
        }

     });
}

function filterStationDataToScheme(schemeName){
    let result = [];

    for (var k in stationDataList) {
        // use hasOwnProperty to filter out keys from the Object.prototype
        if (stationDataList.hasOwnProperty(k))
        {
            const station = stationDataList[k];

            if (station.hasOwnProperty('metaData')){

                if (station.metaData.hasOwnProperty('schemeName'))
                {
                    const scheme = station.metaData.schemeName;
                    if (scheme.toLowerCase() === schemeName.toLowerCase())
                    {
                        result.push(station);
                    }
                }
            }
        }
    }

    return result;

}

function sortStoragesByDamThenWeirThenPercentageFull(storages)
{
    let sortedStorages = storages.sort(function(a, b)
        {
            //dams should factor higher than weirs
            if (a.metaData.isDam && !b.metaData.isDam)
            { return -1; }

            if (!a.metaData.isDam && b.metaData.isDam)
            { return 1; }

            if (!a.hasOwnProperty('currentCapacity')){
                return -1;
            }
            if (!b.hasOwnProperty('currentCapacity')){
                return 1;
            }


            //both are same storage type, so now order by percentage full
            if (a.currentCapacity.percentageFull > b.currentCapacity.percentageFull)
            {
                return -1;
            } 
            else {
                return 1;
            }
        }
    );

    return sortedStorages;

}

{

async function getSites(stationCodes)
{

    // Dams return all dams with no continuation token, so no need to do paging
    var url = 'https://data.sunwater.com.au/api/Sites';
    let dams = await executePOST(url, stationCodes);
    return dams.value;

}

function executeGET(url)
{
    return new Promise((resolve, reject) => {
    
        jQuery.ajax({
            headers: { 
                'Accept': 'application/json',
                'Content-Type': 'application/json' 
            },
            'type': 'GET',
            'url': url,        
            'dataType': 'json',
            'success': function(response){
    
                if (response)
                {
                    resolve(response);
                }
                else {
                    reject("Could not load data");
                }
                
    
            }
            
        });
    });    
}


function getCurrentStorageCapacityForStations(stations)
{

    // Dams return all dams with no continuation token, so no need to do paging
    return new Promise((resolve, reject) => {
        var url = 'https://data.sunwater.com.au/api/StorageCapacity/current';

        resolve(executePOST(url, stations));
    });    
}

function executePOST(url, data)
{
 
    return new Promise((resolve, reject) => {
    
        
        jQuery.ajax({
            headers: { 
                'Accept': 'application/json',
                'Content-Type': 'application/json' 
            },
            'type': 'POST',
            'url': url,
            'data': JSON.stringify(data),
            'dataType': 'json',
            'success': function(response){
    
    
                if (response)
                {
                    resolve(response);
                }
                else {
                    reject("Could not load data");
                }
                
    
            }
            
        });
    });    
}

friendlyDateFull = function(dt) {

    var mm = dt.getMonth() + 1; // getMonth() is zero-based
    var dd = dt.getDate();
    var hr = dt.getHours();
    var min = dt.getMinutes();    
    var ampm = hr >= 12 ? 'pm' : 'am';
    
    if (hr > 12)
    {
        hr = hr - 12;
    }
  
    return [
        hr,
        ':',
        (min < 10 ? '0' : '') + min,
        ampm,
        ' ',
        (dd>9 ? '' : '0') + dd,
        '.',
        (mm>9 ? '' : '0') + mm,
        '.',
        dt.getFullYear(),
    ].join('');
};

friendlyDate = function(dt) {
    
    var mm = dt.getMonth() + 1; // getMonth() is zero-based
    var dd = dt.getDate();    
  
    return [  
        (dd>9 ? '' : '0') + dd,
        '.',
        (mm>9 ? '' : '0') + mm,
        '.',
        dt.getFullYear(),
    ].join('');
};
